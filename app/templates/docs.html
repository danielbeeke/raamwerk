The main concept is conditions and reactions:

You tell the framework if the route is 'about', load the layout two columns with specific blocks.

  • Route is a context condition plugin.
  • Layout is a context reaction plugin.
/app/contexts.js: 

'about': {
  conditions: {
    route: ['about']
  },
  reactions: {
    layer: [{
      key: 'about',
      layout: 'two-columns',
      data: function () {
        return {
          columnOne: 'here your html for column one',
          columnTwo: 'here your html for column two'
        }
      },
      postRender: function () {
        $('.button').click(function () {
          alert('test')
        })
      }
    }]
  }
}
